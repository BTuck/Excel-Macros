Sub ReName()
Dim orignal As String, result As String
'Loops through all X Range'
For Each Cel In Range("b1:b23") 'ActiveCell.CurrentRegion.Cells"
orignal = Cel.Value
'Change NPM'
If orignal = "Mitel NPM" Then
result = "Mitel NuPoint"
'Change MAS'
ElseIf orignal = "MITEL MAS" Then
result = "MAS"
'Change MCD'
ElseIf orignal = "MITEL MCD(3300)" Then
result = "MiVoice Business"
'Change MBG'
ElseIf orignal = "MITEL MBG(6000)" Then
result = "Mitel Border Gateway"
Else
    result = orignal
End If
Cel.Value = result
Next
End Sub
